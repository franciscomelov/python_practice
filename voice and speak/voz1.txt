def takeCommand():
    r = sr.Recognizer()

    with sr.Microphone() as source:

        print('Escuchando...')
        r.pause_threshold = 1
        audio = listener.listen(source)
        command = listener.recognize_google(audio, language="es-ES")
        print(command)

    try:
        print("Recognizing...")
        query = command
        print(f"User said: {query}\n")

    except Exception as e:
        print(e)
        print("Unable to Recognize your voice.")
        return "None"

    return query
